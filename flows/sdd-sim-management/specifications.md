// KI programming for multi-SIM adapters

function program_ki_type9($dongle, $ki, $imsi, $iccid) {
    // Type 9 (STK) adapter programming
    // Uses AT+STGR, AT+STIN commands
    
    // 1. Enter programming mode
    send_at_cmd($dongle, "AT+STSF=1");
    
    // 2. Send KI programming sequence
    $apdu = build_ki_apdu($ki, $imsi, $iccid);
    send_at_cmd($dongle, "AT+CSIM=" . strlen($apdu) . ",\"$apdu\"");
    
    // 3. Exit programming mode
    send_at_cmd($dongle, "AT+STSF=0");
    
    return verify_ki($dongle, $imsi);
}

function program_ki_type16($dongle, $ki, $imsi, $iccid) {
    // Type 16 (SuperSIM) adapter programming

### modules/sim.php - Balance/Stats

<?php
function save_balance($imsi, $balance) {
    $file = "/var/simbox/sim/statistics/{$imsi}.balance";
    file_put_contents($file, $balance);
}

function upload_balance($imsi, $balance) {
    save_balance($imsi, $balance);
    
    $url = "http://simserver:8122/balance.php";
    $params = http_build_query([
        'imsi' => $imsi,
        'balance' => $balance,
        'timestamp' => time()
    ]);
    @file_get_contents("{$url}?{$params}");
