### extensions_dial.conf (excerpt)

```asterisk
; Main outbound dial logic
[default-dongle-outgoing]

; Extract special mode from dialed number
; e.g., +79991234567#VIP -> number=+79991234567, mode=VIP
exten => _X.#*,1
    Set(NUMBER=${CUT(EXTEN,#,1)})
    Set(MODE=${CUT(EXTEN,#,2)})
    
    GotoIf($["${MODE}" = "VIP"]?vip-routing,1)
    GotoIf($["${MODE}" = "SOU"]?sound-test,1)
    GotoIf($["${MODE}" = "NAV"]?nav-mode,1)
    
    ; Normal routing
    Goto(zone-routing,1)

; VIP routing - any SIM
[vip-routing]
exten => s,1
    Set(DONGLE_GROUP=all_work)
    Goto(dial-via-gsm,1)

; Sound test mode
[sound-test]
exten => s,1
    Playback(beep)
    Hangup()

; Zone routing
[zone-routing]
exten => s,1
    Include => extensions-tele2-spb
    Include => extensions-bee-spb
    Include => extensions-meg-spb
    ; ... other zones
    
; Dial via GSM
[dial-via-gsm]
exten => s,1
    Set(CALL_START=${EPOCH})
    Dial(Dongle/${DONGLE}/${NUMBER},60)
    
    ; Check result
    same => n,GotoIf($["${DIALSTATUS}" = "ANSWER"]?success)
    same => n,GotoIf($["${DIALSTATUS}" = "BUSY"]?busy)
    same => n,GotoIf($["${DIALSTATUS}" = "NOANSWER"]?no-answer)
    same => n,GotoIf($["${DIALSTATUS}" = "CHANUNAVAIL"]?chan-unavail)
    
    ; Retry logic
    same => n,Set(FAIL_COUNT=${INC(FAIL_COUNT)})
    same => n,GotoIf($[${FAIL_COUNT} < 3]?retry)
    
    ; All retries failed
    same => n,Playback(all-circuits-busy)
    same => n,Hangup()
```

### extensions_incoming.conf (excerpt)

```asterisk
; Incoming call handling
[default-dongle-incoming]

exten => s,1
    ; Notify central server
    same => n,Set(NUMBERB=${CALLERID(num)})
    same => n,Set(NUMBERMY=${DONGLE_NUMBER})
    same => n,System(
        curl -s "http://simserver:8122/in.php?
        numberb=${NUMBERB}&
        numbermy=${NUMBERMY}&
        imei=${IMEI}&
        imsi=${IMSI}&
        dongle=${DONGLE}&
        uid=${UNIQUEID}"
    )
    
    ; Process call
    same => n,Macro(incoming-full)
```

### system/makecall.sh

```bash
#!/bin/bash
DONGLE=$1
NUMBER=$2
IMSI=$3

CALLFILE="/var/spool/asterisk/outgoing/call.$$"

cat > "$CALLFILE" << EOF
Channel: Dongle/$DONGLE/$NUMBER
MaxRetries: 0
RetryTime: 60
WaitTime: 45
Context: default-dongle-outgoing
Extension: s
Priority: 1
Set: IMSI=$IMSI
Set: DONGLE=$DONGLE
EOF

# Trigger call
mv "$CALLFILE" /var/spool/asterisk/outgoing/
```
