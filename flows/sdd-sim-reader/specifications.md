### wts.pas - Main Application

```pascal
program WTS;
{ Wert SIM-Card Scanner }
{ Version 1.0.6 }

uses SynaSer, SysUtils, Tables, WTypes, Functions;

const
    VERSION = '1.0.6';
    PORTNAME = '/dev/ttyS0';
    PORTSPEED = 9600;

var
    Port: TBlockSerial;
    ResKi: TData16;
    IMSI: String[18];
    ICCID: String[22];

procedure WriteHelp;
begin
    LWriteln('--device=reader     : Using CardReader (Dejian, USI 2.0)', 2);
    LWriteln('--port=<name>        : Device port name', 2);
    LWriteln('--speed=<number>     : Port speed [9600, 19200, 38400]', 2);
    LWriteln('--imsi               : Read IMSI', 2);
    LWriteln('--gsm                : Run GSM-algo (find KI)', 2);
end;

begin
    // Initialize port
    Port := TBlockSerial.Create;
    Port.Connect(PORTNAME, PORTSPEED);
    
    // Reset SIM
    Port.SendString(#$BF #$20);
    
    // Read ATR
    ATR := Port.RecvPacket(1000);
    
    // Read IMSI
    IMSI := ReadIMSI(Port);
    
    // Read ICCID
    ICCID := ReadICCID(Port);
    
    // Extract KI (Comp128v1)
    ResultKI := FindKI(Port, ResKi);
    
    // Save results
    SaveToFile(ICCID, IMSI, ResKi);
end.
```
