### modules/at.php - AT Command Interface

```php
<?php
function send_at_cmd($dongle, $cmd) {
    $escaped_cmd = escapeshellarg($cmd);
    $output = shell_exec(
        "asterisk -rx \"dongle cmd $dongle $escaped_cmd\" 2>&1"
    );
    
    if (strpos($output, "ERROR") !== false) {
        error_log("AT command failed for $dongle: $cmd");
        return null;
    }
    
    return trim($output);
}

function change_imei($dongle, $imei) {
    $result = send_at_cmd($dongle, "AT+CGSN");
    $old_imei = trim($result);
    
    // Calculate NCK
    $nck = calculate_nck($imei);
    
    // Send unlock command
    send_at_cmd($dongle, "AT+NCK=$nck");
    
    // Change IMEI
    send_at_cmd($dongle, "AT+CGSN=$imei");
    
    // Verify
    $new_imei = send_at_cmd($dongle, "AT+CGSN");
    
    return ($new_imei === $imei);
}

function calculate_nck($imei, $type = 1) {
    // NCK calculation algorithm
    $constants = [
        1 => "constant_type_1",
        2 => "constant_type_2"
    ];
    
    $hash = md5($imei . $constants[$type]);
    return substr($hash, 0, 16);  // Simplified
}
?>
```
